    <script>
        var ownIdDataObject = undefined;

        ownid("register", {
            passwordField: document.getElementById("pass"),
            loginIdField: document.getElementById("email"),
            onError: (error) => console.log(error),
            onRegister: (ownIdData) => ownIdDataObject = ownIdData
        });

        function regClick() {
            register(document.querySelector('#email').value, {
                firstName: document.querySelector('#firstName').value,
                lastName: document.querySelector('#lastName').value,
            }, document.querySelector('#pass').value,ownIdDataObject)
        }
    </script>
